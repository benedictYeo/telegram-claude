```typescript
async function handleStatus(chatId: number, env: Env): Promise<void> {
  const usage = getUsage(chatId);
  const model = env.DEFAULT_MODEL || "claude-haiku-4-5-20251001";
  const todayTotal = usage.todayIn + usage.todayOut;
  const totalAll = usage.totalIn + usage.totalOut;
  const lastTs = usage.lastTs
    ? new Date(usage.lastTs * 1000).toISOString().replace("T", " ").substring(0, 16) + " UTC"
    : "---";

  const msg = [
    "<b>Status</b>",
    `Model: <code>${model}</code>`,
    `Today: ${todayTotal} tokens (${usage.todayIn} in / ${usage.todayOut} out)`,
    `Total: ${totalAll} tokens`,
    `Turns: ${usage.turns}`,
    `Last message: ${lastTs}`,
  ].join("\n");

  await tgSend(env, chatId, msg);
}
```
